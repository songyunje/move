DROP DATABASE IF EXISTS ssafy_debug;

CREATE DATABASE ssafy_debug CHARACTER SET=utf8mb4;

USE ssafy_debug;

DROP TABLE IF EXISTS drama;


CREATE TABLE `drama` (
  	`id` INT NOT NULL AUTO_INCREMENT,
    `title` VARCHAR(100) NOT NULL,
    `screenwriter` VARCHAR(40) NOT NULL,
    `director` VARCHAR(40) NOT NULL,
    `actors` VARCHAR(300) NOT NULL,
    PRIMARY KEY(`id`)
)ENGINE=InnoDB;
